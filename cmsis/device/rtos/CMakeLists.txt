if(${MBED_TOOLCHAIN} STREQUAL "GCC_ARM")
    target_sources(mbed-rtos-obj
        PRIVATE
            TOOLCHAIN_GCC_ARM/mbed_boot_gcc_arm.c
    )
elseif(${MBED_TOOLCHAIN} STREQUAL "ARM")
    target_sources(mbed-rtos-obj
        PRIVATE
            TOOLCHAIN_ARM_STD/mbed_boot_arm_std.c
    )
endif()

target_include_directories(mbed-rtos
    INTERFACE
        include
)

target_sources(mbed-os-obj
    PRIVATE
        source/mbed_boot.c
        source/mbed_rtos_rtx.c
        source/mbed_rtx_handlers.c
        source/mbed_rtx_idle.cpp
)

target_compile_definitions(mbed-rtos
    INTERFACE
        MBED_CONF_RTOS_PRESENT=1
)

target_include_directories(mbed-rtos INTERFACE $<TARGET_PROPERTY:${MBED_TARGET_CONVERTED},INTERFACE_INCLUDE_DIRECTORIES>)


target_link_libraries(mbed-rtos-obj PRIVATE mbed-core)

# Link related core files (=mbed-os-obj) that require rtos to get their flags
target_link_libraries(mbed-os-obj PRIVATE mbed-core mbed-rtos)
